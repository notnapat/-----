
- if ใช้กำหนดเงื่อนไข เช่น

if(Statement)
{
 //  เงื่อนไขเป็นจริง
}
else
{
 //  เงื่อนไขเป็นเท็จ
}


- try catch ใช้สำหรับดักจับข้อผิดพลาด และเมื่อมี Error โปรแกรมจะยังสามารถทำงานต่อไปได้ โดยจะกระโดด หรือ โยน error ไปให้ หรือ ไปทำในส่วนของ catch 
-  หรือ คือ catch จะสมารถรันได้แค่ error จาก try เท่านั้น

try {
 //  คำสั่งการทำงาน ปกติ มี if-else ได้
} catch(error) {
 //  ทำงานเมื่อมี Error จาก try
}
